##인코딩 방법
import pandas as pd # pandas가져온다

# 테스트를 위한 데이터프레임을 생성한다 / weight, height, sex, blood_type, health을 column으로 하여 값을 지정해준다
df = pd.DataFrame({'weight':[40, 80, 60, 50, 90], 
                   'height':[162, 155, 182, 173, 177],
                   'sex':['f', 'm', 'm', 'f', 'm'],
                   'blood_type':['O', 'A', 'B', 'O', 'A'],
                   'health':['good', 'excellent', 'bad', 'bad', 'good'],
                   })
df

# ordinalencoder는 범주형 데이터를 정수로 인코딩하는 기법이며 한번에 여러 피처(독립변수)에 사용가능하다
from sklearn.preprocessing import OrdinalEncoder # ordinal encoder라는 클래스를 가져온다

df_oe = df.copy() # 위에서 만들어 놓은 df를 복사하여 df_oe로 저장한다

oe = OrdinalEncoder() # 가져온 OrdinalEncoder클래스로 oe라는 객체를 생성한다

# df_oe 데이터를 만들어 놓은 oe객체로 학습(.fit)시킨다
oe.fit(df) # .fit()함수 사용함

print(f'{oe.categories_=}') # 결과를 보여준다

# 이번에는 실제로 적용(transform)하여 df에 있던 데이터들을 모두 정수로 변환시킨다
df_oe = pd.DataFrame(oe.transform(df), columns=df.columns)
df_oe

# 위에서 사용한 OrdinalEncoder의 다른 방법을 소개한다(특정 부분만 발췌하여 적용하는 방법)
df_oe = df.copy()
oe = OrdinalEncoder()
oe.fit(df[['sex', 'blood_type']]) # 이번에는 범주형 데이터 중 sex와 blood_type만 훈련(fit)시킨다
print(f'{oe.categories_=}') # 결과

# 위에서 훈련시킨 결과를 그 특정 열(sex, blood_type)들에만 삽입한다
df_oe.iloc[:,2:4] = oe.transform(df[['sex', 'blood_type']])
df_oe

oe.inverse_transform(df_oe.iloc[:,2:4]) # 위에서 인코딩한 결과를 원상복구(디코딩)한다


# LabelEncoder라는 클래스를 가져온다 / 이 클래스는 범주형 데이터를 정수로 인코딩하는 또다른 기법이다(단, 위의 OrdinalEncoder과 다르게 한번에 한 독립변수에만 적용이 가능함)
from sklearn.preprocessing import LabelEncoder

df_le = df.copy()

health_le = LabelEncoder().fit(df.health) # 가져온 LabelEncoder()클래스로 health_le라는 객체를 생성하며 동시에 훈련(fit)시킨다
df_le['health'] = health_le.transform(df.health) # .transform()이라는 함수를 사용하여 기존 df_le의 범주형 컬럼이었던 health컬럼만 정수형으로 인코딩함
df_le


# TargetEncoder 클래스를 가져온다 / 이는 범주형 데이터를 수치형 데이터로 인코딩하는 기법중 하나인데, 이 때 변경된 범주형 데이터의 결과가 타겟으로 삼은 다른 수치형 컬럼의 값의 크기와 비례한 숫자가 된다. 
from sklearn.preprocessing import TargetEncoder

df_te = df.copy()

# TargetEncoder클래스를 이용하여 te라는 객체를 생성한다
te = TargetEncoder(smooth=0, target_type='continuous') # smooth = 0은 인코딩을 부드럽게 하는정도를 말하는데, 여기서 부드럽다는 것은 참조하는 다른 컬럼을 딱 1개로만 설정하는것이 전혀 부드럽지 않음(smooth = 0)이고 다른 컬럼들도 좀 혼합해서 사용하는것이 부드러운것(smooth값이 0보다 큼)이다. / target_type = 'continuous'는 타겟으로 삼을 컬럼의 타입을 설정하는것인데, continuous는 연속형 범주라는 의미이다


te.fit(df['blood_type'].values.reshape(-1, 1), df.weight) # df의 blood_type열을 인코딩할 것이며 .values는 데이터를 numpy로 가져오라는 뜻이고 .reshape(-1,1)해당 열을 열벡터로 변환하는데 사용된다(여기서 -1은 다른 차원의 크기를 자동으로 계산하라는 뜻이 담겨있다)

print(f'{te.categories_=}') # 결과

# 학습된 결과를 적용(.transform)하여 blood_type_target이라는 새로운 컬럼에 삽입한다
df_te['blood_type_target'] = te.transform(df['blood_type'].values.reshape(-1, 1))
df_te


# onehotencoder라는 클래스를 가져온다 / onehotencoder는 범주형 데이터로 구성된 컬럼을 수치형 데이터중 하나인 이진수(0또는 1)로 인코딩하는 기법이다
from sklearn.preprocessing import OneHotEncoder

# 아래의 과정은 위와 동일하므로, 중복된 설명은 생략하도록 한다
df_ohe = df.copy()

ohe = OneHotEncoder().fit(df_ohe[['blood_type']]) # 가져온 클래스를 객체로 변환함과 동시에 blood_type 컬럼을 학습시킨다

print(f'{ohe.categories_=}')

# OneHotEncoder는 결과를 희소행렬(sparse matrix)로 반환하므로 toarray()를 통해 numpy배열로 변환
# 여기서 희소행렬이란 대부분의 원소가 0인 행렬을 말한다(많은 0이 존재하는 행렬을 메모리낭비의 최소화하면서 나타낸다)
df_ohe[ohe.categories_[0]] = ohe.transform(df_ohe[['blood_type']]).toarray()
df_ohe

# 위의 onehotencoding과 동일한 기능이다 / pandas에서 제공하며 여러 컬럼을 한번에 인코딩할 수 있고, 회귀분석에서 범주형 변수를 고려할 때 사용된다
pd.get_dummies(df, columns=['sex', 'blood_type'], drop_first=False)

# binarizer라는 클래스를 불러온다 / binarizer는 연속적인 수치형 데이터를 어떠한 기준값을 사용하여 이진수로 인코딩하는 기법이다
from sklearn.preprocessing import Binarizer

# 위와 동일한 코드의 반복이므로, 중복된 설명은 생략한다
df_bin = df.copy()

df_bin['weight_bin'] = Binarizer(threshold=50).fit_transform(df.weight.values.reshape(-1,1)) # threshold = 50은 사용할 임계값이 50이라는 의미이다 / weight열을 인코딩한다
df_bin['height_bin'] = Binarizer(threshold=170).fit_transform(df.height.values.reshape(-1,1))
# binarizer는 연속 수치형 변수에 대해서만 변환 가능하다
df_bin


# LabelBinarizer클래스를 가져온다 /  LabelBinarizer는 연속형 데이터를 이진데이터로 변경하는 다른 방법이다(이는 위의 binarizer기법과 다르게, 문자형 데이터도 인코딩이 가능하다 / 한번에 하나의 컬럼만 인코딩 가능하다)
from sklearn.preprocessing import LabelBinarizer

# 위와 동일한 코드의 반복이므로 중복된 설명은 생략한다
df_lb = df.copy()

lb = LabelBinarizer().fit(df.health) # df의 health컬럼을 인코딩한다
print(f'{lb.classes_ = }')
health_lb = lb.transform(df.health)
print('health_lb = \n', health_lb)

# 인코딩된 결과는 health가 good이면 good열에 그 특정행에 '1'값이 입력되고, health가 bad면 bad열에 대응되는 행에 '1'값이 입력됨
# 총 bad, excellent, good이라는 3개의 열이 새로 생김과 동시에 이진수가 입력됨 / 이를 LabelBinarizer에서의 인코딩이라고 부름
df_lb[lb.classes_] = health_lb
df_lb


# MultiLabelBinerizer 클래스를 가져옴 / 여러개의 범주(하나의 데이터에 math,science가 입력되어 있다는 의미)가 있는 데이터를 이진수 컬럼으로 인코딩하는 기법
from sklearn.preprocessing import MultiLabelBinarizer

# 아래의 코드중, 중복된 설명은 생략한다
df_mlb = df.copy()

# test열을 새로 만들고 여러개의 데이터를 생성함
df_mlb['test'] = [['math', 'english'], ['math', 'science'], ['science'], ['math', 'english'], 
                           ['science']]
df_mlb

mlb = MultiLabelBinarizer().fit(df_mlb.test)
print(f'{mlb.classes_ = }')

df_mlb[mlb.classes_] = mlb.transform(df_mlb.test)
df_mlb

## scikit-learn으로 pca와 lda수행하기
from sklearn import datasets # sklearn 라이브러리에서 datasets 모듈을 가져온다
from sklearn.decomposition import PCA  # sklearn 라이브러리의 서브 모듈인 decomposition에서 pca라는 클래스를 가져온다 / pca는 주성분 분석을 하는데 사용한다
from sklearn.discriminant_analysis import LinearDiscriminantAnalysis # sklearn 라이브러리의 서브모듈인 discriminant_analysis에서 LinearDiscriminantAnalysis라는 클래스를 가져온다 / LinearDiscriminantAnalysis는 선형판별분석(lda)을 하는데 사용한다 

# scikit-learn에서의 iris데이터를 가져온다
iris = datasets.load_iris() # 이 scikit-learn에 내장되어 있는 iris데이터는 데이터프레임이 아니라 클래스인 특징을 갖고 있다 


X = iris.data # iris 데이터셋의 피쳐(독립변수 x)들을 x에 저장한다 / 이 x는 150개의 행과 4개의 열을 가지고 있으며, 이는 4개의 독립변수를 가진 데이터 150개가 있다는 것을 의미한다
y = iris.target # iris 데이터셋의 타겟(종속변수 y)들을  y에 저장한다 / 이 y는 setosa, versicolor, virginica라는 세개의 종류로 150개의 데이터가 존재하는 것이다. 그리고 이는 인덱스로 표현된다(setosa는 [0], versicolor는 [1], virginica는 [2])
target_names = list(iris.target_names) # iris 데이터셋의 타겟 이름(setosa, versicolor, virginica)을 리스트로 만들어서 target_names에 저장한 것이다

print(f'{X.shape = }, {y.shape = }') # x,y변수의 행,열의 갯수를 보여준다
print(f'{target_names = }')  # y에 인덱스값으로 변환되어 들어가 있는 타겟이름을 보여준다

pca = PCA(n_components=2) # 위에서 가져온 pca클래스로 pca객체를 만든다 / n_components = 2 는 주성분의 개수가 2개라는 의미이며 또한 pca과정을 통해 만들 차원이 2차원이라는 의미도 갖고 있다.

pca_fitted = pca.fit(X) # .fit()함수를 이용하여 위에서 만들어 놓은 x데이터에 pca과정을 진행한다 / 또한 pca는 비지도학습이므로, y데이터값이 필요없기에 x데이터만 입력한 것이다

print(f'{pca_fitted.components_ = }')  # 저장한 pca_fitted의 주성분 벡터를 프린트하여 보여준다 / 위에서 "n_components = 2"이렇게 설정한 대로 총 2개의 차원이 나온다
print(f'{pca_fitted.explained_variance_ratio_ = }') # 각 주성분 벡터가 설명가능한 분산 비율을 프린트하여 보여준다

X_pca = pca_fitted.transform(X) # pca에 의해 학습된 주성분벡터를 사용하여 x데이터를 새로운 좌표(2차원)으로 축소하여 x_pca에 저장한다
print(f'{X_pca.shape = }')  # x_pca데이터의 행, 열 개수를 보여줌으로서 2차원으로 축소된 것을 확인시켜준다

lda = LinearDiscriminantAnalysis(n_components=2) # 위에서 가져온 lda클래스로 lda객체를 만든다 / 마찬가지로 n_components = 2 는 주성분 개수가 2개라는 의미이며, lda과정을 통해 만들 차원이 2차원이라는 의미이다

lda_fitted = lda.fit(X, y) # .fit()함수를 사용하여 lda작업을 수행한다 / lda는 지도학습이므로, x데이터와 함께 결과데이터(타겟값)인 y데이터도 같이 넣어줘야 한다

print(f'{lda_fitted.coef_=}') # 만들어 놓은 lda_fitted에서 .coer를 사용하여 LDA의 계수를 보여준다
print(f'{lda_fitted.explained_variance_ratio_=}') # 위의 pca작업과 마찬가지로 이번엔 LDA의 각 계수가 설명가능한 분산 비율을 보여준다

X_lda = lda_fitted.transform(X) # lda에 의해 학습된 계수를 사용하여 x데이터를 새로운 차원(2차원)으로 축소하여 x_lda에 저장한다
print(f'{X_lda.shape = }')  # 위 pca과정과 마찬가지로 x_lda데이터의 행, 열 개수를 보여줌으로서 2차원으로 축소된 것을 확인시켜준다

from sklearn import datasets # 위에서 설명한것과 동일하게 datasets모듈을 가져온다
from sklearn.feature_selection import VarianceThreshold # sklearn 라이브러리의 서브모듈인 feature_Selection에서 variancethreshold라는 클래스를 불러온다

iris = datasets.load_iris() # 위와 마찬가지로 불러온 datasets모듈에 저장되어 있는 iris데이터를 .load_iris()함수를 이용하여 iris객체로 저장한다

X = iris.data 
y = iris.target

sel = VarianceThreshold(threshold=0.2).fit(X) # 위에서 불러온 variancethreshold클래스를 이용하여 sel이라는 객체를 만드는 과정이다 / threshold = 0.2는 임계값이 0.2라는 의미이며 분산이 0.2이상인 독립변수(피쳐)들만 선택하라는 뜻이다. / 이 또한 .fit()함수를 활용하여 학습시켰고, 필터기법이기에 y데이터(타겟데이터)가 필요 없으므로, x데이터만 입력하였다.
print(f'{sel.variances_ = }') # 각 독립변수(피쳐)의 분산을 보여준다

X_selected = sel.transform(X) # 위에서 저장한 sel객체에 .transform()함수를 사용하여 값을 실제로 적용하고 이를 x_selected에 저장한다
X_selected_names = [X_names[i] for i in sel.get_support(indices=True)] # 선택된 피쳐들의 이름을 가져오는 코드이다 / for 반복문을 사용하여 간단하게 표현하였으며, sel객체의 .get_support()함수를 사용하여 선택된 특성의 인덱스 이름을 반환한다.(정확히는 불리언마스크를 반환하는 것인데, 이는 특정 조건을 만족하면 true, 만족하지 못하면 false를 반환한다. 그렇기에 indices = True를 적어주어서 true인 인덱스들만 반환하여 불러오는 것이다) / 불러온 이름들은 리스트의 형태로 담아진다.  

print(f'{X_selected_names = }') # 선택된 독립변수(피쳐)들의 이름을 보여준다

from sklearn.datasets import load_iris # sklearn라이브러리의 하위모듈인 datasets에서 load_iris라는 함수를 가져온다
from sklearn.feature_selection import RFE, RFECV, SelectFromModel, SequentialFeatureSelector # 마찬가지로 RFE, RFECV, SelectFromModel, SequentialFeatureSelector 함수들을 가져온다
from sklearn.svm import SVC, SVR # 마찬가지로 SVC, SVR 함수들을 가져온다

X, y = load_iris(return_X_y=True) # return_x_y = True는 iris데이터를 가져올 떄 독립변수인 피쳐와 타겟데이터를 모두 가져오라는 의미이다 / 그리고 각각을 x,y에 저장한다
 
svc = SVR(kernel="linear", C=3) # 가져온 svr()함수로 svc라는 객체를 생성한다(이때 svc는 회귀를 위한 서포트벡터머신을 의미한다.))/ kernel = 'linear'은 svm이 사용할 경계를 선형으로 설정한다는 의미이고, c=3에서 c는 정규화 매개변수라고 불리우는데, 이 c값이 커질수록 오차에 더욱 민감해지고 작아질수록 오차에 덜 민감해진다.(오차에 더 민감해진다는 의미는 다른말로 설명하자면 앞으로 주어질 훈련데이터를 더 정확하게 구분하려고 한다는 뜻이기도 하다 / c가 0.1과 같은 작은값이면 오차에 덜 민감하고 10,100과같은 큰 값이면 오차에 매우 민감해져서 모델의 과적합가능성이 존재한다)
# 서포트 벡터 머신 : 분류 기법중 하나인 supprot vector mechine의 약어인데, 주어진 데이터의 margin을 설정하고 이 margin안쪽에 오는 데이터들을 기준으로 한 오차를 최소화 하여 분류하는 기법

rfe = RFE(estimator=svc, n_features_to_select=2, step=1) # 가져온 rfe()라는 함수로 rfe라는 객체를 생성한다(rfe기법은 서포트 벡터 머신을 활용하여 피처들을 순차적으로 제거하면서 모델의 성능을 평가하는 방법(래퍼기법)이다) / estimator = svc는 rfe가 사용할 모델을 서포트벡터 머신으로 하겠다는 의미이다. n_features_to_select=2는 선택하려는 최종 피처의 개수를 2개로 지정한다는 의미이다. step=1은 각 반복단계에서 제거할 피처의 개수가 1개라는 의미이다.
rfe_cv = RFECV(estimator=svc, step=1, cv=5) # 가져온 rfecv()라는 함수를 이용하여 rfe_cv라는 객체를 생성한다 / estimator = svc와 step = 1은 모두 위의 설명과 같고 cv=5는 교차검증에서 데이터셋을 나누는 부분의 수(폴드)를 5개로 설정한다는 의미이다.

## 파이프라인
# 파이프라인이란, 여러가지 함수들을 하나의 pipeline으로 연결하여 함수형태로 만든 뒤 적용을 한꺼번에 하는 방법이다
# 파이프라인을 사용하면, 변환된 데이터를 매번 저장하지 않고 연속적으로 한번에 사용하기에, 속도가 개선되며 메모리도 절약됨


# 아래는 파이프라인을 사용하지 않는 경우이다 / 피처를 선택하며, 표준화하고, 모델학습까지 하는 과정을 할때마다 입력해줘야 해서 불편함이 있다
# 아래의 코드가 주요내용이 아니므로 설명은 생략하도록 한다
from sklearn.feature_selection import SelectKBest, f_classif 
from sklearn.preprocessing import StandardScaler
from sklearn.tree import DecisionTreeClassifier
from sklearn.datasets import load_iris

# iris 데이터세트 로드
X, y = load_iris(return_X_y=True)
 
# 피쳐 선택
feat_sel = SelectKBest(f_classif, k=2)
X_selected = feat_sel.fit_transform(X, y)
print('Selected features:', feat_sel.get_feature_names_out())

# 표준화
scaler = StandardScaler()
scaler.fit(X_selected)
X_transformed = scaler.transform(X_selected)
print('Standard Scaled: \n', X_transformed[:5, :])

# 모델 학습
clf = DecisionTreeClassifier(max_depth=3)
clf.fit(X_transformed, y)
print('Estimate : ', clf.predict(X_transformed)[:3])
print('Accuracy : ', clf.score(X_transformed, y))


# 지금부터가 파이프라인을 사용한 경우이다 / 아래에 피처선택이나 데이터표준화, 분류기법등등은 위의 피처엔지니어링에서 자세히 설명하였으므로, 중복된 설명은 생략한다 
from sklearn.pipeline import Pipeline # pipeline이라는 클래스를 가져온다
from sklearn.feature_selection import SelectKBest, f_classif
from sklearn.preprocessing import StandardScaler
from sklearn.tree import DecisionTreeClassifier
from sklearn.datasets import load_iris
 
# iris 데이터세트를 가져와서 x,y에 저장함
X, y = load_iris(return_X_y=True)
 
# 아래는 pipeline 구축하는 과정이다
pipeline = Pipeline([
    ('Feature_Selection', SelectKBest(f_classif, k=2)), # 피쳐 선택은 f_classif기법을 사용한다
    ('Standardization', StandardScaler()),  # 표준화는 standardscaler기법을 사용한다
    ('Decision_Tree', DecisionTreeClassifier(max_depth=3)) # 분류기법은 의사결정나무를 활용한다 / max_depth = 3은 의사결정나무의 leaf를 만들어 가는 과정에서 깊이를 3으로 설정한다는 의미임
])
# 구성한 파이프라인을 확인한다
display(pipeline)

# 만들어놓은 파이프라인을 위에서 가져온 iris데이터로 학습(.fit())시킨다
pipeline.fit(X, y)

# 예측(추정)값 제시 및 성능평가(정확도)를 보여준다
print('Estimate : ', pipeline.predict(X)[:3])
print('Accuracy : ', pipeline.score(X, y))


# sklearn의 pipeline클래스에서 make_pipeline라는 파이프라인 구성을 위한 함수를 가져온다
# make_pipeline은 위의 pipeline과 다르게 파이프라인의 이름을 자동적으로 만들어 준다(만들어진 파이프라인의 이름은 사용하는 클래스들의 이름을 소문자로 변경한것임)
from sklearn.pipeline import make_pipeline

pipeline_auto = make_pipeline(SelectKBest(f_classif, k=2), 
              StandardScaler(), 
              DecisionTreeClassifier(max_depth=3))
display(pipeline_auto)


# 파이프라인의 피처선택 결과의 중간과정을 확인
pipeline.named_steps['Feature_Selection'] == pipeline[0]
pipeline.named_steps['Standardization'] == pipeline[1]
pipeline.named_steps['Decision_Tree'] == pipeline[2]
print('Selected features:', pipeline.named_steps['Feature_Selection'].get_feature_names_out())
X_transformed = pipeline[1].transform(X_selected)
print('Standard Scaled: \n', X_transformed[:5, :])


## 파이프라인의 결합
import seaborn as sns
import pandas as pd
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import StandardScaler
from sklearn.impute import SimpleImputer

# seaborn의 diamond 데이터를 가져온다
df = sns.load_dataset('diamonds')
print(df.info())
X = df.drop('price', axis=1) # x데이터에는 가져온 diamonds데이터에서 price데이터를 떼고 저장한다 / price데이터는 타겟값으로 사용하기 위함임
y = df['price'] # df의 price만 타겟데이터로 저장함

# 데이터를 유형(범주형 / 수치형)에 따라 분리한다
numeric_col = list(X.select_dtypes(exclude='category').columns) # .select_dtypes()함수를 사용하여 x데이터에서의 특정 데이터타입만 고른다 / exclude = category는 범주형데이터만 제외하고 리스트로 변환 후 numeric_col에 저장한다 / .columns는 x데이터에서 각각의 열들을 대상으로 앞의 select_dtype함수를 적용한다는 뜻이다
category_col = list(X.select_dtypes(include='category').columns) # 위와 동일하나 include = category는 범주형 데이터만 선택한다는 의미이다
# 결과보여줌
print(f'numeric_col: {numeric_col}')
print(f'category_col: {category_col}')

# 파이프라인을 구축한다 / imputer과정과 scaler과정을 합친것인데, 변환은 nan값을 평균값으로 대체하는것으로 하고 / 스케일링은 표준화(평균0, 표준편차 1)과정을 거친다
numeric_pipeline = Pipeline(
    steps=[
        ('imputer', SimpleImputer(strategy='mean'))
        ('scaler', StandardScaler())
    ])

# 파이프라인 확인
display(numeric_pipeline)

# 만든 파이프라인을 학습과 동시에 적용(.fit_transform())시키고 이를 numerical_data_piped에 저장한다
numerical_data_piped = numeric_pipeline.fit_transform(X[numeric_col]) # 위에서 만들어놓은 x데이터를 이용하여 학습시킨다 / x[numeric_col]은 x데이터 중 수치형데이터에 해당하는것만 가져온다는 뜻임
# 저장한 numerical_data_piped를 데이터프레임으로 바꾼 뒤 처음 5개행만 보여준다
pd.DataFrame(numerical_data_piped, columns=numeric_col).head()


# 범주형 데이터 파이프라인 처리
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import OneHotEncoder

# 범주형 데이터관련 파이프라인을 구축한다 / 값을 대체하는 과정(SimpleImputer)과 인코딩(OneHotEncoder)하는 과정을 파이프라인으로 만듬
category_pipeline = Pipeline(
    steps=[
        ('imputer', SimpleImputer(strategy='constant', fill_value='missing')), # strategy = constant는 상수값으로 대체하라는 의미이며 / fill_value = missing은 비어있는 값을 'missing'으로 채우라는 뜻이다
        ('onehot', OneHotEncoder(sparse_output=False)), # 범주형 데이터를 이진수(0또는 1)로 인코딩하는 기법인 onehotencoder를 사용한다 / sparse_output=False는 희소행렬이 아닌 밀집배열로 출력하라는 뜻이다
    ])
# 위에서 밀집배열이란 희소행렬과는 반대로 대부분의 원소가 0이 아닌 값을 갖는 행렬을 말한다

# 파이프라인 확인
display(category_pipeline)

# 만든 범주형 데이터 파이프라인을 학습시킨다
category_data_piped = category_pipeline.fit_transform(X[category_col]) # 이번에는 x데이터에서 범주형값들만 가져옴

# .get_feature_names_out()함수를 이용하여 Onehotencoder에 의해 인코딩된 컬럼명들을을 확인한다
category_colnames = category_pipeline[1].get_feature_names_out(category_col) # category_pipeline[1]은 만든 파이프라인에서 두번째과정을 가져온다는 의미이다 /  

# 적용시킨 파이프라인(category_data_piped)데이터를 데이터프레임으로 만든 뒤 처음 5개행을 가져옴
pd.DataFrame(category_data_piped, columns=category_colnames).head()


# 수치형과 범주형을 동시에 처리할 수 있는 파이프라인을 만든다
from sklearn.compose import ColumnTransformer # 서로다른 유형의 데이터를 전처리하기 위해 columntransformer클래스를 가져옴
from sklearn.linear_model import LinearRegression # 선형회귀방법을 사용하기 위해 linearregression라는 클래스를 가져옴

# 수치형과 범주형 파이프라인을 합친다 / columntransformer클래스를 이용함
preprocessor = ColumnTransformer(
    transformers=[
        ('numeric', numeric_pipeline, numeric_col),
        ('category', category_pipeline, category_col)
    ])

pipe = make_pipeline(preprocessor, LinearRegression()) # 만든 preprocessor에다가 선형회귀분류기법을 추가하여 pipe라는 파이프라인을 만든다
# 만들어진 최종 파이프라인 확인
display(pipe)
# .fit()함수로 만든 최종 파이프라인을 학습시킨다
pipe.fit(X,y)

print('Estimate : ', pipe.predict(X))
print('Accuracy : ', pipe.score(X, y))


# 위에서 사용한 columntransformer를 다른 예시를 들어 설명한다
from sklearn.compose import ColumnTransformer
from sklearn.impute import SimpleImputer
import pandas as pd
import numpy as np

# 수치형과 범주형데이터를 모두 갖춘 데이터프레임(data_df)을 생성한다
data_df = pd.DataFrame({
    "height":[165,  np.nan, 182],
    "weight":[70,   62,     np.nan],
    "age"   :[np.nan,18,    15]
})

# SimpleImputer를 사용해서 height의 null 값들은 평균으로 출력하고 나머지 열들은 통과시키는 파이프라인을 만들어 col_transformer에 저장한다
col_transformer = ColumnTransformer([
    ("Impute_mean", SimpleImputer(strategy="mean"), ["height"])
    ], 
    remainder="passthrough"
)

# 만든 파이프라인 그래프로 구성 확인
display(col_transformer)
# 위에서 만든 data_df를 보여준다
print(data_df)
# 만든 col_transformer 파이프라인을 활용하여 data_df를 학습시킴과 동시에 적용한다(.fit_transform())
print(col_transformer.fit_transform(data_df))

# 이번에는 SimpleImputer를 사용해서 height의 null값을 평균으로 대체하고 weight열의 null값을 중앙값으로 대체한다 
# 그 외의 나머지 열의 null값은 상수값(strategy = 'constant')인 '-1'로 대체한다
col_transformer2 = ColumnTransformer([
    ("Impute_mean"  , SimpleImputer(strategy="mean")  , ["height"]),
    ("Impute_median", SimpleImputer(strategy="median"), ["weight"])
    ],
    remainder=SimpleImputer(strategy="constant", fill_value=-1)
)

# 만든 파이프라인을 확인한다
display(col_transformer2)

print(data_df)
# 만든 col_transformer2 파이프라인을 활용하여 data_df를 학습시킴과 동시에 적용한다
print(col_transformer2.fit_transform(data_df))
